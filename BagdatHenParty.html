  async function polishWish(name, text) {

    const prompt = `You are a warm, poetic assistant helping guests leave beautiful wishes for a bride named Bagdat at her bachelorette party.


A guest named "${name}" wrote:

"${text}"


Rewrite it as a single short, heartfelt, slightly poetic message (2–4 sentences). Keep the original sentiment. Make it warm, celebratory and elegant. Return ONLY the polished wish text — nothing else.`;


    try {

      const res = await fetch('https://api.anthropic.com/v1/messages', {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify({

          model: 'claude-sonnet-4-20250514',

          max_tokens: 200,

          messages: [{ role: 'user', content: prompt }]

        })

      });

      const data = await res.json();

      const txt = data.content && data.content[0] && data.content[0].text;

      if (txt) return { polished: txt.trim(), aiEnhanced: true };

    } catch (_) {}

    return { polished: text, aiEnhanced: false };

  }
